4.Creating and querying views and materialized views.

syntax:
create or replace force/noforce view viewname as 
select column_list from table_list
where condition with read only|check option
Dropping a view syntax:
Drop view viewname;


create table views
(
rollno integer,
name varchar2(20),
marks integer
);

create view myview as select rollno,name from views;
desc myview;

select * from tab;

insert into myview values(503,'prathisha');

select * from myview;


select * from views;

insert into views values(507,'kamesh',50);

select * from myview;

select * from views


delete from myview where rollno=503;
 
select * from myview;

select * from views
