3.Find sailors who have not reserved any boat.

select * from sailors s
where s.sid not in (select r.sid from reserves r);

select * from sailors s
where not exists(select * from reserves r where r.sid=s.sid);



4.Find sailors who are older than every sailor with a rating of 10.

select * from sailors
where age > ALL (select age from sailors where rating = 10);

select * from sailors
where age > (select max(age) from sailors where rating = 10);
