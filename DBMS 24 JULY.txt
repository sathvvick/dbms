create table suppliers
(
sid integer,
sname varchar2(20) not null,
address varchar2(30),
primary key(sid)
);

create table parts
(
pid integer,
pname varchar2(20) not null,
color varchar2(30),
primary key(pid)
);

create table catalogs
(
sid integer,
pid integer,
cost number(7,2),
primary key(sid,pid),
foreign key(sid) references suppliers(sid),
foreign key(pid) references parts(pid));
